@app.route('/robots.txt')
def robots():
    """Fichier robots.txt SEO-friendly et dynamique"""
    domain = AppConfig.DOMAIN.rstrip("/")

    content = (
        "User-agent: *\n"
        "Allow: /\n"
        "Allow: /sitemap.xml\n"
        "Allow: /mentions-legales\n"
        "Allow: /politique-confidentialite\n"
        "Allow: /conditions-utilisation\n"
        "Allow: /contact\n"
        "Allow: /a-propos\n"
        "\n"
        "Disallow: /admin/\n"
        "Disallow: /uploads/\n"
        "Disallow: /temp/\n"
        "\n"
        f"Sitemap: https://{domain}/sitemap.xml\n"
        "\n"
        "# PDF Fusion Pro - Développé par MYBSOW\n"
        "# Contact: banousow@gmail.com\n"
    )

    return Response(content, mimetype="text/plain")
