// ── Boutons cloud — CORRECTION ──────────────────────────────────────────────
// Les boutons utilisent data-cloud="..." au lieu d'onclick inline.
// On attache les listeners ici, après que cloud-upload.js est chargé (il est
// inclus dans base.html avant ce bloc). En cas d'échec de chargement du script,
// un fallback ouvre directement l'URL du service dans un nouvel onglet.
document.addEventListener('DOMContentLoaded', function () {
    var CLOUD_FALLBACK_URLS = {
        google:   'https://drive.google.com',
        onedrive: 'https://onedrive.live.com',
        dropbox:  'https://www.dropbox.com',
        icloud:   'https://www.icloud.com',
        box:      'https://www.box.com',
        pcloud:   'https://www.pcloud.com'
    };

    document.querySelectorAll('[data-cloud]').forEach(function (btn) {
        btn.addEventListener('click', function () {
            var provider = this.getAttribute('data-cloud');
            if (typeof window.cloudUpload !== 'undefined') {
                // Script chargé correctement → comportement normal
                window.cloudUpload.open(provider);
            } else {
                // Fallback : ouverture directe dans un nouvel onglet
                var url = CLOUD_FALLBACK_URLS[provider];
                if (url) {
                    window.open(url, '_blank');
                }
                console.warn('[cloud] cloudUpload non disponible, ouverture directe de', provider);
            }
        });
    });
});
